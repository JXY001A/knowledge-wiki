```
    ├──→ memory/reader.py

    │       ├── search_memory()     — FTS5 全文搜索

    │       └── recent_context()    — LLM 上下文注入
    │
    ├──→ memory/working.py
    │       └──→ skills/query-knowledge/impl.py — 注入记忆上下文
    │
    └──→ memory/profile.py
            └──→ wiki/USER.md — 用户画像自动维护
```
